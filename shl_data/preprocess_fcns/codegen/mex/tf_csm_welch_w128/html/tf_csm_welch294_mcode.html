1%% Function for spectra estimation by Welch's method
  2% Developed by Luiz A. Baccala, Fl?vio Caduda and Luciano Caldas, all from
  3% Escola Polit?cnica - Poli-USP, with cooperation of Carlos Pagani and Felipe
  4% Amaral from Escola de Engenharia de S?o Carlos - EESC-USP.
  5%
  6% Cross-spectra matrix are estimated by Welch's method with 50% overlap and
  7% the window energy loss are compasated by a factor of 1/sum(Wi.^2) where
  8% Wi are the elements of the window [1]. Then, the spectra becomes:
  9% Sxy = fft(x)*conj(fft(y))/sum(Wi.^2)
 10%
 11% Code was tested with a known- spectra signal from a white noise filtered
 12% by a filter. The variance (power) of the signal checks with the integral
 13% of the PSD estimated.
 14%
 15% INPUT:
 16% -- signals: matrix of signals to perform the spectra estimatino. Size is
 17% [Samples x number of sensors];
 18% -- fs: samplerate in Hertz;
 19% -- window: data taper desired. Must be a vector. For best performance it
 20% should be a power of 2. For general applications do: window=hanning(1024);
 21%
 22% OUTPUT:
 23% -- CSM: Cross Spectral Matrix: Unilateral (0:fs/2) spectra. Welch's
 24% method is used with 50% overlap. Matrix size: sensors x sensors x
 25% windowsize/2
 26% -- frequencies: vector with all frequencies corresponding to each layer
 27% (3rd layer in depth) of CSM.
 28%
 29% LAST REVISION: Aug - 18 - 2016
 30% ADDED 'fs' missing term in line 82, for calibration factor
 31% [1] Trobs,M.; Heinzel,G. "Improved spectrum estimation from digitized
 32% time series on a logarithmic frequency axis"
 33% doi:10.1016/j.measurement.2005.10.010
 34function [CSM,frequencies] = tf_csm_welch(signals,fs,window)
 35if size(signals,2) > size(signals,1)
 36    signals = signals.';
 37end
 38window = window(:);
 39sensors = size(signals,2);
 40windowsize = length(window);
 41frequencies = (0:(windowsize/2-1))*fs/windowsize;
 42block_samples = windowsize; %must be even, best if 2^n
 43signal_samples = size(signals,1);
 44number_of_signals = size(signals,2);
 45back_shift = block_samples./2; %ORIGINAL;
 46number_of_blocks = floor((2*signal_samples)./block_samples) - 1;
 47data_taper = window;
 48data_taper = repmat(data_taper,1,number_of_signals);
 49% Data segmentation into blocks of size block_samples:
 50if number_of_signals == 1
 51    S = zeros(block_samples/2,number_of_signals.^2); % FOR SINGLE SIGNAL
 52else
 53    S = zeros(block_samples/2,number_of_signals.^2 - 1); %ORIGINAL
 54end
 55
 56for a = 1:number_of_blocks
 57    % Retrieve current data block
 58    Data_Block = signals((a-1)*back_shift+1:block_samples +(a-1)*back_shift,:);
 59    Data_Block = Data_Block - repmat(mean(Data_Block),block_samples,1); % Mean-shift (zeroing)
 60    Data_Block = Data_Block.*data_taper; %Taper it
 61    Data_Block = fft(Data_Block); %FFT it,
 62    % bilateral DFT
 63    % viii
 64    Data_Block = Data_Block(1:block_samples/2,:); %ORIGINAL
 65    % Data_Block = Data_Block(1:ceil(block_samples/2),:);
 66    %All spectral combinations:
 67    P = zeros(block_samples/2,number_of_signals.^2 - 1); %ORIGINAL
 68    c = 1;
 69    for aa = 1:size(Data_Block,2)
 70        for b = aa:size(Data_Block,2)
 71            % P(:,c) = Data_Block(:,b).*conj(Data_Block(:,aa)); % THIS
 72            % IS FOR WIND TUNNEL EESC-USP BEAMFORMING CODE
 73%             P(:,c) = Data_Block(:,aa).*conj(Data_Block(:,b)); % THIS IS THE ORIGINAL
 74            % LINE
 75            P(:,c) = real(Data_Block(:,aa).*conj(Data_Block(:,b)));
 76%             IS FOR FAN RIG BEAMFORMING CODE
 77            c = c+1;
 78        end
 79    end
 80    % Sum the spectrums up ...
 81    S = S + P;
 82end
 83S = S*2/(sum(window.^2)*fs*number_of_blocks); % Average them out
 84CSM = S; 
 85% Sf = zeros(sensors,sensors,size(S,1));
 86
 87% c=1;
 88% for a = 1:sensors
 89%     for b = a:sensors
 90%         Sf(a,b,:) = S(:,c);
 91%         c = c+1;
 92%     end
 93% end
 94% clear S
 95% CSM = Sf;
 96% Goes through n x n x : matrix and does:
 97% = ORIGINAL + TRANSPOSE - IDENTITY(ORIGINAL). 
 98% for i = 1:size(CSM,3)
 99%     CSM(:,:,i) = CSM(:,:,i) + CSM(:,:,i)' - eye(sensors).*CSM(:,:,i);
100% end
101% 
102% CSM = reshape(CSM, [size(CSM, 1)*size(CSM,2), size(CSM, 3)]);
103
104end
