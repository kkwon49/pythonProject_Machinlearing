  489     //  Sxy = fft(x)*conj(fft(y))/sum(Wi.^2)
  490     //
  491     //  Code was tested with a known- spectra signal from a white noise filtered
  492     //  by a filter. The variance (power) of the signal checks with the integral
  493     //  of the PSD estimated.
  494     //
  495     //  INPUT:
  496     //  -- signals: matrix of signals to perform the spectra estimatino. Size is
  497     //  [Samples x number of sensors];
  498     //  -- fs: samplerate in Hertz;
  499     //  -- window: data taper desired. Must be a vector. For best performance it
  500     //  should be a power of 2. For general applications do: window=hanning(1024); 
  501     //
  502     //  OUTPUT:
  503     //  -- CSM: Cross Spectral Matrix: Unilateral (0:fs/2) spectra. Welch's
  504     //  method is used with 50% overlap. Matrix size: sensors x sensors x
  505     //  windowsize/2
  506     //  -- frequencies: vector with all frequencies corresponding to each layer
  507     //  (3rd layer in depth) of CSM.
  508     //
  509     //  LAST REVISION: Aug - 18 - 2016
  510     //  ADDED 'fs' missing term in line 82, for calibration factor
  511     //  [1] Trobs,M.; Heinzel,G. "Improved spectrum estimation from digitized
  512     //  time series on a logarithmic frequency axis"
  513     //  doi:10.1016/j.measurement.2005.10.010
  514     //  frequencies = (0:(windowsize/2-1))*fs/windowsize;
  515     // must be even, best if 2^n
  516     // ORIGINAL;
  517     repmat(window, data_taper);
  518   
  519     //  Data segmentation into blocks of size block_samples:
  520     // ORIGINAL
  521     //  S = zeros(ceil(block_samples/2),number_of_signals.^2);
  522     //  Retrieve current data block
  523     a = mean(signals);
  524   
  525     // Taper it
  526     for (i = 0; i < 256; i++) {
  527       b_signals[i] = (signals[i] - a) * data_taper[i];
  528     }
  529   
  530     fft(b_signals, Data_Block);
  531   
  532     // FFT it,
  533     //  bilateral DFT
  534     //  viii
  535     // ORIGINAL
  536     //  Data_Block = Data_Block(1:ceil(block_samples/2),:);
  537     // All spectral combinations:
  538     // ORIGINAL
  539     //  P = zeros(ceil(block_samples/2)/2,number_of_signals.^2);
  540     //  THIS IS FOR WIND TUNNEL EESC-USP BEAMFORMING CODE
  541     //              P(:,c) = real(Data_Block(:,b).*conj(Data_Block(:,aa)));
  542     //  P(:,c) = Data_Block(:,b).*conj(Data_Block(:,aa));
  543     //  IS FOR FAN RIG BEAMFORMING CODE
  544     //  P(:,c) = Data_Block(:,aa).*conj(Data_Block(:,b)); % THIS IS THE ORIGINAL LINE 
  545     //  Sum the spectrums up ...
  546     power(window, dv7);
  547     a = sum(dv7) * fs;
  548   
  549     //  Average them out
  550     //  for a = 1:sensors
  551     for (i = 0; i < 128; i++) {
  552       CSM[i] = (Data_Block[i].re * Data_Block[i].re - Data_Block[i].im *
  553                 -Data_Block[i].im) * 2.0 / a;
  554     }
  555   
  556     //  end
  557     //  clear S
  558     CSM[0] = (CSM[0] + CSM[0]) - CSM[0];
  559   }
