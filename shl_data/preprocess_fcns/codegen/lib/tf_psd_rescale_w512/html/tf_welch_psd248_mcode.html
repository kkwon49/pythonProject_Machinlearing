1%% Function for spectra estimation by Welch's method
 2% Developed by Luiz A. Baccala, Fl?vio Caduda and Luciano Caldas, all from
 3% Escola Polit?cnica - Poli-USP, with cooperation of Carlos Pagani and Felipe
 4% Amaral from Escola de Engenharia de S?o Carlos - EESC-USP.
 5%
 6% Cross-spectra matrix are estimated by Welch's method with 50% overlap and
 7% the window energy loss are compasated by a factor of 1/sum(Wi.^2) where
 8% Wi are the elements of the window [1]. Then, the spectra becomes:
 9% Sxy = fft(x)*conj(fft(y))/sum(Wi.^2)
10%
11% Code was tested with a known- spectra signal from a white noise filtered
12% by a filter. The variance (power) of the signal checks with the integral
13% of the PSD estimated.
14%
15% INPUT:
16% -- signals: matrix of signals to perform the spectra estimatino. Size is
17% [Samples x number of sensors];
18% -- fs: samplerate in Hertz;
19% -- window: data taper desired. Must be a vector. For best performance it
20% should be a power of 2. For general applications do: window=hanning(1024);
21%
22% OUTPUT:
23% -- CSM: Cross Spectral Matrix: Unilateral (0:fs/2) spectra. Welch's
24% method is used with 50% overlap. Matrix size: sensors x sensors x
25% windowsize/2
26% -- frequencies: vector with all frequencies corresponding to each layer
27% (3rd layer in depth) of CSM.
28%
29% LAST REVISION: Aug - 18 - 2016
30% ADDED 'fs' missing term in line 82, for calibration factor
31% [1] Trobs,M.; Heinzel,G. "Improved spectrum estimation from digitized
32% time series on a logarithmic frequency axis"
33% doi:10.1016/j.measurement.2005.10.010
34function [CSM] = tf_welch_psd(signals,fs,window)
35% if size(signals,2) > size(signals,1)
36%     signals = signals.';
37% end
38signals = signals(:);
39window = window(:);
40sensors = size(signals,2);
41windowsize = length(window);
42% frequencies = (0:(windowsize/2-1))*fs/windowsize;
43block_samples = windowsize; %must be even, best if 2^n
44signal_samples = size(signals,1);
45number_of_signals = size(signals,2);
46back_shift = block_samples./2; %ORIGINAL;
47number_of_blocks = floor((2*signal_samples)./block_samples) -1;
48data_taper = window;
49data_taper = repmat(data_taper,1,number_of_signals);
50% Data segmentation into blocks of size block_samples:
51S = zeros(block_samples/2,number_of_signals.^2); %ORIGINAL
52% S = zeros(ceil(block_samples/2),number_of_signals.^2);
53for a = 1:number_of_blocks
54    % Retrieve current data block
55    Data_Block = signals((a-1)*back_shift+1:block_samples +(a-1)*back_shift,:);
56    Data_Block = Data_Block - repmat(mean(Data_Block),block_samples,1);
57    Data_Block = Data_Block.*data_taper; %Taper it
58    Data_Block = fft(Data_Block); %FFT it,
59    % bilateral DFT
60    % viii
61    Data_Block = Data_Block(1:block_samples/2,:); %ORIGINAL
62    % Data_Block = Data_Block(1:ceil(block_samples/2),:);
63    %All spectral combinations:
64    P = zeros(block_samples/2,number_of_signals.^2); %ORIGINAL
65    % P = zeros(ceil(block_samples/2)/2,number_of_signals.^2);
66    c = 1;
67    for aa = 1:size(Data_Block,2)
68        for b = aa:size(Data_Block,2)
69            % THIS IS FOR WIND TUNNEL EESC-USP BEAMFORMING CODE
70%             P(:,c) = real(Data_Block(:,b).*conj(Data_Block(:,aa))); 
71            % P(:,c) = Data_Block(:,b).*conj(Data_Block(:,aa)); 
72            % IS FOR FAN RIG BEAMFORMING CODE
73            P(:,c) = real(Data_Block(:,aa).*conj(Data_Block(:,b)));
74            % P(:,c) = Data_Block(:,aa).*conj(Data_Block(:,b)); % THIS IS THE ORIGINAL LINE
75            c = c+1;
76        end
77    end
78    % Sum the spectrums up ...
79    S = S + P;
80end
81S = S*2/(sum(window.^2)*fs*number_of_blocks); % Average them out
82Sf = zeros(sensors,size(S,1));
83c=1;
84% for a = 1:sensors
85    for b = 1:sensors
86        Sf(b,:) = S(:,c);
87        c = c+1;
88    end
89% end
90% clear S
91CSM = Sf;
92for i = 1:size(CSM,3)
93    CSM(:,i) = CSM(:,i) + CSM(:,i)' - eye(sensors).*CSM(:,i);
94end
95
96end
