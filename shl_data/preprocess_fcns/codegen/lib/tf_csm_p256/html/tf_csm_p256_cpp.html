  411     //  Function for spectra estimation by Welch's method
  412     //  Developed by Luiz A. Baccala, Fl?vio Caduda and Luciano Caldas, all from
  413     //  Escola Polit?cnica - Poli-USP, with cooperation of Carlos Pagani and Felipe 
  414     //  Amaral from Escola de Engenharia de S?o Carlos - EESC-USP.
  415     //
  416     //  Cross-spectra matrix are estimated by Welch's method with 50% overlap and
  417     //  the window energy loss are compasated by a factor of 1/sum(Wi.^2) where
  418     //  Wi are the elements of the window [1]. Then, the spectra becomes:
  419     //  Sxy = fft(x)*conj(fft(y))/sum(Wi.^2)
  420     //
  421     //  Code was tested with a known- spectra signal from a white noise filtered
  422     //  by a filter. The variance (power) of the signal checks with the integral
  423     //  of the PSD estimated.
  424     //
  425     //  INPUT:
  426     //  -- signals: matrix of signals to perform the spectra estimatino. Size is
  427     //  [Samples x number of sensors];
  428     //  -- fs: samplerate in Hertz;
  429     //  -- window: data taper desired. Must be a vector. For best performance it
  430     //  should be a power of 2. For general applications do: window=hanning(1024); 
  431     //
  432     //  OUTPUT:
  433     //  -- CSM: Cross Spectral Matrix: Unilateral (0:fs/2) spectra. Welch's
  434     //  method is used with 50% overlap. Matrix size: sensors x sensors x
  435     //  windowsize/2
  436     //  -- frequencies: vector with all frequencies corresponding to each layer
  437     //  (3rd layer in depth) of CSM.
  438     //
  439     //  LAST REVISION: Aug - 18 - 2016
  440     //  ADDED 'fs' missing term in line 82, for calibration factor
  441     //  [1] Trobs,M.; Heinzel,G. "Improved spectrum estimation from digitized
  442     //  time series on a logarithmic frequency axis"
  443     //  doi:10.1016/j.measurement.2005.10.010
  444     // must be even, best if 2^n
  445     // ORIGINAL;

  447     number_of_blocks = (int)std::floor(z);

  449   
  450     //  Data segmentation into blocks of size block_samples:

  452

  454     for (a = 0; a <= number_of_blocks - 2; a++) {
  455       //  Retrieve current data block
  456       c = a << 7;
  457       for (i0 = 0; i0 < 2; i0++) {
  458         for (aa = 0; aa < 256; aa++) {
  459           Data_Block[aa + (i0 << 8)] = signals_data[(aa + c) + signals_size[0] *

  461         }
  462       }

  464       mean(Data_Block, dv2);
  465       b_repmat(dv2, dv3);
  466   
  467       //  Mean-shift (zeroing)

  478       // ORIGINAL
  479       //  Data_Block = Data_Block(1:ceil(block_samples/2),:);
  480       // All spectral combinations:
  481       memset(&P[0], 0, 384U * sizeof(double));
  482   
  483       // ORIGINAL
  484       c = 0;
  485       for (aa = 0; aa < 2; aa++) {
  486         for (b = 0; b <= 1 - aa; b++) {
  487           //  P(:,c) = Data_Block(:,b).*conj(Data_Block(:,aa)); % THIS
  488           //  IS FOR WIND TUNNEL EESC-USP BEAMFORMING CODE
  489           //              P(:,c) = Data_Block(:,aa).*conj(Data_Block(:,b)); % THIS IS THE ORIGINAL 
  490           //  LINE
  491           b_aa = aa + b;
  492           for (i0 = 0; i0 < 128; i0++) {
  493             P[i0 + (c << 7)] = b_Data_Block[i0 + (aa << 8)].re * b_Data_Block[i0 +

  496           }
  497   
  498           //              IS FOR FAN RIG BEAMFORMING CODE
  499           c++;
  500         }
  501       }
  502   
  503       //  Sum the spectrums up ...
  504       for (i0 = 0; i0 < 384; i0++) {
  505         S[i0] += P[i0];
  506       }
  507     }
  508   
  509     power(window, dv1);
  510     z = sum(dv1) * 250.0 * (std::floor(z) - 1.0);
  511   
  512     //  Average them out
  513     for (i0 = 0; i0 < 384; i0++) {
  514       CSM[i0] = (float)(S[i0] * 2.0 / z);
  515     }
  516   
  517     for (c = 0; c < 3; c++) {
  518       memcpy(&b_CSM[0], &CSM[c << 7], sizeof(float) << 7);
  519       rescale_minmax(b_CSM, *(float (*)[128])&CSM[c << 7]);
  520     }
  521   
  522     //  Sf = zeros(sensors,sensors,size(S,1));
  523     //  c=1;
  524     //  for a = 1:sensors
  525     //      for b = a:sensors
  526     //          Sf(a,b,:) = S(:,c);
  527     //          c = c+1;
  528     //      end
  529     //  end
  530     //  clear S
  531     //  CSM = Sf;
  532     //  Goes through n x n x : matrix and does:
  533     //  = ORIGINAL + TRANSPOSE - IDENTITY(ORIGINAL).
  534     //  for i = 1:size(CSM,3)
  535     //      CSM(:,:,i) = CSM(:,:,i) + CSM(:,:,i)' - eye(sensors).*CSM(:,:,i);
  536     //  end
  537     //
  538     //  CSM = reshape(CSM, [size(CSM, 1)*size(CSM,2), size(CSM, 3)]);
